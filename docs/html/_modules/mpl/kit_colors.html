def gray(intensity: float) -> tuple:
    """
    Creates an RGB faded tuple for any gray scale

    Parameters
    ----------
    intensity: float
        gray intensity  with range from 0 (white) to 1 (full color)

    Returns
    -------
    tuple
    """
    return (1-intensity, 1-intensity, 1-intensity)

def hex_to_faded(hex_color: str, intensity: float = 1.) -> tuple:
    """
    Creates an RGB faded tuple to reduce color intensity

    Parameters
    ----------
    hex_color: str
        hex code of color
    intensity: float
        color intensity  with range from 0 (white) to 1 (full color)

    Returns
    -------
    tuple
    """
    # Convert hex color to RGB tuple
    hex_color = hex_color.strip('#')
    rgb_color = tuple(int(hex_color[i:i + 2], 16) for i in (0, 2, 4))

    # Calculate faded RGB tuple
    faded_rgb = tuple(((1-intensity) + intensity * c/255) for c in rgb_color)

    # Return faded RGB tuple as argument tuple
    return faded_rgb + (1,)
